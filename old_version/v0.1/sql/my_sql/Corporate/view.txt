CREATE VIEW Corporate.v_list
AS
SELECT 
	C.code, replace(C.name,'&','¾Ø') name, C.market,
	P.close price, P.close - P.uprisePrice yesterday, P.uprisePrice uprisePrice,
	(P.uprisePrice) / (P.close - P.uprisePrice) upriseRatio
FROM 
	Corporate.list C, Price.v_candle P
WHERE
	P.date = (SELECT max(date) FROM Price.techIndex) AND
	C.code = P.code;
